        alphabets="abcdefghijklmnopqrstuvwxyz"

        for j in range(len(c)):

          r.append(a1)
        y=''.join(r)

        for m in range(len(d)):
            a2 =alphabets[alphabets.index(d[m])+(k*(-1))]
            s.append(a2)
        z=''.join(s)
        z=z.capitalize()
        p = x + " " + y + " " + z

        u=math.pow(10,3)
